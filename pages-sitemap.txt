import { getServerSideSitemap } from 'next-sitemap'
import { getPayload } from 'payload'
import config from '@payload-config'
import { unstable_cache } from 'next/cache'

const normalizeUrl = (url: string) => url.replace(/\/+$/, '')

const getPagesSitemap = unstable_cache(
  async () => {
    const payload = await getPayload({ config })
    const SITE_URL = normalizeUrl(
      process.env.NEXT_PUBLIC_SERVER_URL ||
        process.env.VERCEL_PROJECT_PRODUCTION_URL ||
        'https://webdesigngrid.com'
    )

    const results = await payload.find({
      collection: 'pages',
      overrideAccess: false,
      draft: false,
      depth: 0,
      limit: 1000,
      pagination: false,
      where: {
        _status: {
          equals: 'published',
        },
      },
      select: {
        slug: true,
        updatedAt: true,
        breadcrumbs: true,
      },
    })

    const dateFallback = new Date().toISOString()
    console.log('site url', SITE_URL)
    
    // Build sitemap using reduce - starting with static pages
    const sitemap = (results.docs || []).reduce(
      (acc, page) => {
        // Filter out invalid pages
        if (!page?.slug) return acc

        // Single expression URL generation - no nested conditionals
        const pageUrl =
          page.breadcrumbs?.[page.breadcrumbs.length - 1]?.url ||
          (page.slug === 'web-design-agency' ? '/' : `/${page.slug}`)

        // Add page to accumulator
        acc.push({
          loc: `${SITE_URL}${pageUrl}`,
          lastmod: page.updatedAt || dateFallback,
        })

        return acc
      },
      // Initialize accumulator with static pages
      [
        {
          loc: `${SITE_URL}/search`,
          lastmod: dateFallback,
        },
        {
          loc: `${SITE_URL}/blogs`,
          lastmod: dateFallback,
        },
      ]
    )

    return sitemap
  },
  ['pages-sitemap'],
  {
    tags: ['pages-sitemap'],
  },
)

export async function GET() {
  const sitemap = await getPagesSitemap()
  return getServerSideSitemap(sitemap)
}
